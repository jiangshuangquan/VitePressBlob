  name: 'ParallaxBtn',
  data() {
    return {
      rx: '0deg',
      ry: '0deg',
      tx: '0',
      ty: '0',
      tz: '0px'
    };
  },
  methods: {
    mousemove(e) {
      const btn = document.getElementsByClassName('parallax-btn')[0];
      const btnRect = btn.getBoundingClientRect();
      const dx = e.offsetX - btnRect.width / 2;
      const dy = e.offsetY - btnRect.height / 2;
      this.rx = `${dy / -1}deg`;
      this.ry = `${dx / 10}deg`;
    },
    mouseup(e) {
      this.tz = '-12px';
    },
    mousedown(e) {
      this.tz = '-25px';
    },
    mouseleave(e) {
      this.rx = `0`;
      this.ry = `0`;
      this.ty = `0`;
    }
  }
};
